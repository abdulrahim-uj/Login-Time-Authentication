CREATE OR REPLACE PROCEDURE SCOTT.PROC_GSA_TEST (CASE_NO NUMBER, TIME_TS IN varchar2, PRESS_TIME IN NUMBER, GSA_ID_NO IN VARCHAR2)
    IS
    GSA_ID1 VARCHAR2(15);
    DD TIMESTAMP(9);
--    CURSOR FETCH_DATA
--        IS
--        SELECT GSA_ID FROM TBL_SIGNUP WHERE TABLE_ID = (SELECT MAX(TABLE_ID) FROM TBL_SIGNUP);
        BEGIN
--        OPEN FETCH_DATA;
--        FETCH FETCH_DATA INTO GSA_ID1;
        IF CASE_NO = 0 THEN
            GSA_ID1 := GSA_ID_NO;
            DBMS_OUTPUT.PUT_LINE('HELLO WORLD GSA_ID_1 0 ' ||GSA_ID1);
            UPDATE TBL_SIGNUP SET TIME_1 =TO_TIMESTAMP(TIME_TS,'YYYY-MM-DD HH24:MI:SS.FF3') WHERE GSA_ID =GSA_ID_NO;
            DBMS_OUTPUT.PUT_LINE('HELLO WORLD GSA_ID 1 ' ||GSA_ID_NO);
            COMMIT;                
            UPDATE TBL_TIMEPROCE SET TIME_1 =PRESS_TIME WHERE GSA_ID = GSA_ID_NO;
            DBMS_OUTPUT.PUT_LINE('HELLO WORLD GSA_ID 2 ' ||GSA_ID_NO);
            COMMIT;
        END IF;
        IF CASE_NO = 1 THEN
            DBMS_OUTPUT.PUT_LINE('HELLO WORLD GSA_ID 1 CASE 1 ' ||GSA_ID_NO);
                UPDATE TBL_SIGNUP SET TIME_2 =TO_TIMESTAMP(TIME_TS,'YYYY-MM-DD HH24:MI:SS.FF3') WHERE GSA_ID =GSA_ID_NO;
                DBMS_OUTPUT.PUT_LINE('HELLO WORLD GSA_ID 1 CASE 1 2 ' ||GSA_ID_NO);
                COMMIT;                
                UPDATE TBL_TIMEPROCE SET TIME_2 =PRESS_TIME WHERE GSA_ID = GSA_ID_NO;
                DBMS_OUTPUT.PUT_LINE('HELLO WORLD GSA_ID 1 CASE 1 3 ' ||GSA_ID_NO);
                COMMIT;
        END IF;
            END;
/
