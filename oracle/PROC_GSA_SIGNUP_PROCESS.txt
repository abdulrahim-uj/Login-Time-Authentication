CREATE OR REPLACE PROCEDURE SCOTT.PROC_GSA_SIGNUP_PROCESS (USER_NAME IN VARCHAR2, U_PASSWORD IN VARCHAR2, TOTAL_TIME IN NUMBER, GSA_ID_NO IN VARCHAR2, U_TYPE IN NUMBER)
IS
    BEGIN
       DBMS_OUTPUT.PUT_LINE('HELLO  1' || GSA_ID_NO);
       UPDATE TBL_SIGNUP SET UNAME = USER_NAME, PWD = U_PASSWORD, TYPE = U_TYPE, TOTAL_TIME_TAKEN = TOTAL_TIME
       WHERE GSA_ID = GSA_ID_NO;
       COMMIT;
       DBMS_OUTPUT.PUT_LINE('HELLO  2' || GSA_ID_NO);
       UPDATE TBL_TIMEPROCE SET UNAME = USER_NAME, PWD = U_PASSWORD, TYPE = U_TYPE, TOTAL_TIME_TAKEN = TOTAL_TIME
       WHERE GSA_ID = GSA_ID_NO;
       COMMIT;
       UPDATE TBL_GSA_ID_GENERATORE SET OWNER = USER_NAME
       WHERE GSA_ID = GSA_ID_NO
       AND OWNER IS NULL;
       DBMS_OUTPUT.PUT_LINE('HELLO 3 ' || GSA_ID_NO);
        EXCEPTION
                                                WHEN OTHERS THEN
                                                RAISE_APPLICATION_ERROR(-20001,'AN ERROR WAS ENCOUNTERED -
                                                '||SQLCODE||' -ERROR- '||SQLERRM);
    END;
/
